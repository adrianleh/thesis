                    \State $out.predecessors \gets (out.predecessors \backslash \{phi\}) \cup \newline \{n \vert  n \in phi.predecessors, n.loop = out.loop\}$
                \EndIf
            \EndFor
        \EndFunction
    \end{algorithmic}
