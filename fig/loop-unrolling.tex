                \While{$i < 16$}
                    \State \Call{Print}{$i$}
                    \State \Call{Print}{$i + 1$}
                    \State \Call{Print}{$i + 2$}
                    \State \Call{Print}{$i + 3$}
                    \State $i \gets i + 4$
                \EndWhile
            \EndFunction
        \end{algorithmic}
    \end{minipage}
